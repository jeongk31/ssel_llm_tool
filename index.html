#!/usr/bin/env python3
"""
LLM Measurement Pipeline — MI Communication Coding
Generated by SSELab LLM Measurement Toolkit
Config hash: sha256:a3f8c2...d91e

Usage:
    pip install litellm pandas
    export OPENAI_API_KEY=sk-...
    export ANTHROPIC_API_KEY=sk-ant-...
    export GOOGLE_API_KEY=AI...
    python pipeline_run.py --input data.csv --output results.jsonl
"""

import json
import hashlib
import argparse
from pathlib import Path
from collections import Counter
from concurrent.futures import ThreadPoolExecutor, as_completed

import pandas as pd
import litellm

# ── Configuration ────────────────────────────────────────
TASK_NAME = "MI Communication Coding"
UNIT = "episode"
OUTPUT_TYPE = "classify"

MODELS = [
    {"provider": "openai", "model": "gpt-4o", "weight": 2},
    {"provider": "anthropic", "model": "claude-sonnet-4", "weight": 1},
    {"provider": "google", "model": "gemini-2.0-flash", "weight": 1},
]

RUNS_PER_MODEL = 3
TEMPERATURE = 0.0
SEED = 42
MAX_TOKENS = 1024
CONCURRENCY = 5
MAX_RETRIES = 2

# ── Codebook ─────────────────────────────────────────────
CATEGORIES = {
    "Reflection": "Mirrors or paraphrases client's statement",
    "Open Question": "Exploratory, non-yes/no question",
    "Affirmation": "Acknowledges client strength or effort",
    "Info Giving": "Provides factual info or psychoeducation",
    "Closed Question": "Yes/no or limited-choice question",
    "Confrontation": "Challenges, disagrees, or argues with client",
}

# ── Prompts ──────────────────────────────────────────────
SYSTEM_PROMPT = """You are an expert behavioral coder trained in
psychotherapy process research. Classify therapist utterances into
MI communication categories. Output valid JSON only."""

USER_TEMPLATE = """Classify this therapist utterance into one category.

Categories:
{categories}

Episode: {episode_text}

JSON: {{"label": "<category>", "confidence": <0-100>}}"""

# ── Output Schema ────────────────────────────────────────
OUTPUT_SCHEMA = {
    "type": "object",
    "properties": {
        "label": {"type": "string", "enum": list(CATEGORIES.keys())},
        "confidence": {"type": "integer", "minimum": 0, "maximum": 100},
    },
    "required": ["label", "confidence"],
}


def classify_episode(episode_text: str, model_cfg: dict) -> dict:
    """Send a single episode to a single model and parse the response."""
    categories_str = "\n".join(
        f"- {k}: {v}" for k, v in CATEGORIES.items()
    )
    messages = [
        {"role": "system", "content": SYSTEM_PROMPT},
        {"role": "user", "content": USER_TEMPLATE.format(
            categories=categories_str, episode_text=episode_text
        )},
    ]
    resp = litellm.completion(
        model=model_cfg["model"],
        messages=messages,
        temperature=TEMPERATURE,
        max_tokens=MAX_TOKENS,
        seed=SEED,
    )
    return json.loads(resp.choices[0].message.content)


def weighted_majority(outputs: list[dict], models: list[dict]) -> dict:
    """Aggregate outputs via weighted majority vote."""
    weighted_counts = Counter()
    confidences = {}
    for o in outputs:
        w = next(m["weight"] for m in models if m["model"] == o["model"])
        weighted_counts[o["label"]] += w
        confidences.setdefault(o["label"], []).append(o["confidence"])
    winner = weighted_counts.most_common(1)[0][0]
    avg_conf = sum(confidences[winner]) / len(confidences[winner])
    return {
        "label": winner,
        "confidence": round(avg_conf, 1),
    }


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--input", required=True)
    parser.add_argument("--output", default="results.jsonl")
    args = parser.parse_args()

    df = pd.read_csv(args.input)
    results = []

    with ThreadPoolExecutor(max_workers=CONCURRENCY) as pool:
        for _, row in df.iterrows():
            outputs = []
            for model in MODELS:
                for _ in range(RUNS_PER_MODEL):
                    result = classify_episode(row["content"], model)
                    outputs.append({**result, "model": model["model"]})
            voted = weighted_majority(outputs, MODELS)
            results.append({
                "episode_id": row["episode_id"],
                **voted,
                "model_outputs": outputs,
            })

    Path(args.output).write_text(
        "\n".join(json.dumps(r) for r in results)
    )
    print(f"Done. {len(results)} episodes → {args.output}")


if __name__ == "__main__":
    main()
